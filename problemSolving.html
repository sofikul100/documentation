Problem 1: <-------BUBBLE SORT WITH EXAMPLE IN C# ----> 

    using System;

    class Program
    {
        static void Main()
        {
            int[] arr = { 64, 34, 25, 12, 22, 11, 90 };
            
            Console.WriteLine("Unsorted array:");
            PrintArray(arr);
            
            BubbleSort(arr);
            
            Console.WriteLine("\nSorted array:");
            PrintArray(arr);
        }
    
        static void BubbleSort(int[] arr)
        {
            int n = arr.Length;
            bool swapped;
            
            for (int i = 0; i < n - 1; i++)
            {
                swapped = false;
                
                for (int j = 0; j < n - i - 1; j++)
                {
                    if (arr[j] > arr[j + 1])
                    {
                        // Swap arr[j] and arr[j + 1]
                        int temp = arr[j];
                        arr[j] = arr[j + 1];
                        arr[j + 1] = temp;
                        
                        swapped = true;
                    }
                }
                
                // If no two elements were swapped in the inner loop, then the array is sorted
                if (!swapped)
                    break;
            }
        }
    
        static void PrintArray(int[] arr)
        {
            foreach (int value in arr)
            {
                Console.Write(value + " ");
            }
            Console.WriteLine();
        }
    }

Problem 2: <-------SELECTION SORT WITH EXAMPLE IN C# (BANGLA EXPLAIN) ----> 

Selection Sort হল একটি সরল এবং সহজবোধ্য সর্সিং অ্যালগরিদম। এটি প্রতি ধাপে অ্যারের একটি ছোট উপসেট থেকে সর্বনিম্ন বা সর্বাধিক মান নির্বাচন করে এবং সেটি সঠিক স্থানে রাখে। এটি তুলনামূলকভাবে ধীরগতি সম্পন্ন, তবে বোঝার জন্য সহজ।

Selection Sort অ্যালগরিদম কীভাবে কাজ করে?
প্রথমত, অ্যারের প্রথম এলিমেন্টের সাথে সবগুলো এলিমেন্ট তুলনা করা হয় এবং সবচেয়ে ছোট এলিমেন্টটিকে প্রথম স্থানে আনা হয়।
এরপর অ্যারের বাকি অংশের জন্য একই প্রক্রিয়া পুনরাবৃত্তি করা হয়, অর্থাৎ দ্বিতীয় স্থানে সবচেয়ে ছোট এলিমেন্ট আনা হয়।
এই প্রক্রিয়া চলতে থাকে যতক্ষণ না পুরো অ্যারে সঠিকভাবে সাজানো হয়।
উদাহরণ:
ধরা যাক আমাদের অ্যারে হল [64, 25, 12, 22, 11]। Selection Sort এ কীভাবে এটি কাজ করবে তা নিচে দেখানো হলো:

প্রথমত, [64, 25, 12, 22, 11] থেকে সবচেয়ে ছোট মান নির্বাচন করা হবে। এটি 11, তাই 64 এর জায়গায় 11 আসবে। অ্যারে: [11, 25, 12, 22, 64]

তারপর, 25, 12, 22, 64 এর মধ্যে সবচেয়ে ছোট মান হল 12, তাই 25 এর জায়গায় 12 আসবে। অ্যারে: [11, 12, 25, 22, 64]

এরপর 25, 22, 64 থেকে সবচেয়ে ছোট মান 22, তাই 25 এর জায়গায় 22 আসবে। অ্যারে: [11, 12, 22, 25, 64]

সবশেষে, 25 এবং 64 এর জন্য কোনো অদলবদল করার প্রয়োজন নেই, কারণ তারা সঠিকভাবে সাজানো আছে।

Selection Sort C# কোড:     

    using System;

    class Program
    {
        static void Main()
        {
            int[] arr = { 64, 25, 12, 22, 11 };
            
            Console.WriteLine("অসজ্জিত অ্যারে:");
            PrintArray(arr);
            
            SelectionSort(arr);
            
            Console.WriteLine("\nসজ্জিত অ্যারে:");
            PrintArray(arr);
        }
    
        static void SelectionSort(int[] arr)
        {
            int n = arr.Length;
    
            // প্রতিটি এলিমেন্টের জন্য
            for (int i = 0; i < n - 1; i++)
            {
                // ধরুন, i তম এলিমেন্টের অবস্থানটি সঠিক
                int minIndex = i;
    
                // i+1 থেকে শেষ পর্যন্ত বাকি এলিমেন্টগুলো চেক করা হবে
                for (int j = i + 1; j < n; j++)
                {
                    if (arr[j] < arr[minIndex])
                    {
                        minIndex = j; // ছোট এলিমেন্ট পাওয়া গেলে minIndex আপডেট হবে
                    }
                }
    
                // সবচেয়ে ছোট এলিমেন্টকে i তম স্থানে রাখুন
                int temp = arr[minIndex];
                arr[minIndex] = arr[i];
                arr[i] = temp;
            }
        }
    
        static void PrintArray(int[] arr)
        {
            foreach (int value in arr)
            {
                Console.Write(value + " ");
            }
            Console.WriteLine();
        }
    }

Problem 3: <-------INSERTION SORT WITH EXAMPLE IN C# (BANGLA EXPLAIN) ----> 

Insertion Sort একটি সহজ কিন্তু কার্যকরী সর্সিং অ্যালগরিদম, যা অ্যারের প্রতিটি এলিমেন্টকে তার সঠিক স্থানে বসায়। এটি ধীরে ধীরে এলিমেন্টগুলোকে সাজানো অংশের মধ্যে সঠিক অবস্থানে প্রবেশ করায়, ঠিক যেমন করে একজন কার্ড খেলোয়াড় তার হাতে কার্ড সাজায়।

Insertion Sort কীভাবে কাজ করে?
প্রথমে, অ্যালগরিদম ধরা হয় যে প্রথম এলিমেন্টটি সঠিক স্থানে আছে (কারণ একক এলিমেন্ট স্বাভাবিকভাবেই সাজানো থাকে)।
এরপর, বাকি এলিমেন্টগুলো একে একে নেওয়া হয় এবং তাদের সঠিক স্থানে বসানো হয় সাজানো অংশের মধ্যে।
প্রতিটি ধাপে বর্তমান এলিমেন্টকে পেছনের সাজানো অংশের সাথে তুলনা করা হয় এবং যতক্ষণ না সঠিক অবস্থান পাওয়া যায়, এলিমেন্টগুলোকে ডানে সরিয়ে দেয়া হয়।
পুরো অ্যারে এভাবে সাজানো হয়।
উদাহরণ:
ধরা যাক, আমাদের অ্যারে হলো: [12, 11, 13, 5, 6]

Insertion Sort কীভাবে কাজ করবে তা ধাপে ধাপে দেখা যাক:

ধাপ 1: প্রথম এলিমেন্ট 12 কে নেওয়া হয়। এটি একটি একক এলিমেন্ট হওয়ার কারণে এটি ইতিমধ্যেই সাজানো।
অ্যারে: [12, 11, 13, 5, 6]
ধাপ 2: পরের এলিমেন্ট 11 কে তার সঠিক স্থানে বসানো হয়। এটি 12 থেকে ছোট, তাই 11 কে 12 এর আগে বসানো হয়।
অ্যারে: [11, 12, 13, 5, 6]
ধাপ 3: এরপর 13 কে নেওয়া হয়। এটি তার সঠিক স্থানে রয়েছে কারণ এটি 12 এর পরে আসে।
অ্যারে: [11, 12, 13, 5, 6]
ধাপ 4: 5 কে তার সঠিক স্থানে বসানো হয়। 5 ছোট হওয়ায় এটি 11, 12, 13 এর আগে বসে।
অ্যারে: [5, 11, 12, 13, 6]
ধাপ 5: সবশেষে, 6 কে তার সঠিক স্থানে বসানো হয়। এটি 11 এর আগে, কিন্তু 5 এর পরে বসে।
অ্যারে: [5, 6, 11, 12, 13] 

ব্যাখ্যা (Bangla):
InsertionSort ফাংশন:

প্রথমে অ্যারের দৈর্ঘ্য n বের করা হয়।
বাইরের লুপটি i থেকে শুরু করে দ্বিতীয় এলিমেন্ট থেকে শেষ পর্যন্ত চলে, কারণ প্রথম এলিমেন্ট আগে থেকেই সাজানো আছে।
key দ্বারা প্রতিটি ধাপে বর্তমান এলিমেন্টকে ধরে রাখা হয়।
j দ্বারা পূর্ববর্তী এলিমেন্টগুলোকে চেক করা হয়, যদি তারা key এর চেয়ে বড় হয়, তবে তাদের ডানে সরানো হয়।
সব শেষে key কে তার সঠিক স্থানে বসানো হয়।
PrintArray ফাংশন:

এই ফাংশন অ্যারের সবগুলো এলিমেন্টকে কনসোলে প্রিন্ট করে।
কিভাবে এটি কাজ করে?
প্রতিটি ধাপে বর্তমান এলিমেন্টকে (যাকে key বলা হচ্ছে) তার সঠিক স্থানে বসানোর জন্য পূর্বের এলিমেন্টগুলোর সাথে তুলনা করা হয়।
বড় এলিমেন্টগুলোকে ডানে সরিয়ে দিয়ে সঠিক স্থান বের করা হয়, যেখানে ছোট এলিমেন্টকে বসানো যায়।
সময় জটিলতা:
Worst Case: O(n²)
যদি অ্যারে উল্টানোভাবে সাজানো থাকে (ডিসেন্ডিং অর্ডারে), তাহলে প্রতিটি ধাপে সব এলিমেন্ট চেক করতে হয়, যার ফলে সময় জটিলতা O(n²) হয়।

using System;

class Program
{
static void Main()
{
    int[] arr = { 12, 11, 13, 5, 6 };
    
    Console.WriteLine("অসজ্জিত অ্যারে:");
    PrintArray(arr);
    
    InsertionSort(arr);
    
    Console.WriteLine("\nসজ্জিত অ্যারে:");
    PrintArray(arr);
}

static void InsertionSort(int[] arr)
{
    int n = arr.Length;

    // অ্যারের প্রথম এলিমেন্ট ধরা হয় সাজানো
    for (int i = 1; i < n; i++)
    {
        int key = arr[i]; // বর্তমান এলিমেন্ট নেওয়া হল
        int j = i - 1;

        // বর্তমান এলিমেন্টকে এর সঠিক স্থানে বসানোর জন্য তুলনা করা হচ্ছে
        while (j >= 0 && arr[j] > key)
        {
            arr[j + 1] = arr[j]; // ডানে সরিয়ে নেওয়া হচ্ছে বড় এলিমেন্টগুলো
            j = j - 1;
        }

        arr[j + 1] = key; // সঠিক স্থানে এলিমেন্ট বসানো হচ্ছে
    }
}

static void PrintArray(int[] arr)
{
    foreach (int value in arr)
    {
        Console.Write(value + " ");
    }
    Console.WriteLine();
}
}

Problem 4: <-------Quick SORT WITH EXAMPLE IN C# (BANGLA EXPLAIN) ----> 

Quick Sort-এর কাজের পদ্ধতি
পিভট নির্বাচন: একটি এলিমেন্ট নির্বাচন করুন যাকে পিভট বলা হয়। এটি আপনার সারণীর বিভাজন পয়েন্ট হিসেবে কাজ করবে। সাধারণত, প্রথম, শেষ বা মধ্যবর্তী এলিমেন্ট পিভট হিসেবে নির্বাচন করা হয়।

পার্টিশনিং: সারণীটি এমনভাবে পুনর্বিন্যাস করুন যে পিভটের চেয়ে ছোট সব এলিমেন্ট পিভটের বামে এবং পিভটের চেয়ে বড় সব এলিমেন্ট পিভটের ডানে চলে যাবে।

রিকার্সিভ সোর্টিং: পিভটের বাম এবং ডান পার্টিশনে পুনরায় Quick Sort প্রয়োগ করুন।

স্টপ কন্ডিশন: যখন সারণীটির সাইজ ১ বা ০ হয়ে যাবে, তখন সোর্টিং শেষ হয়ে যাবে।
ব্যাখ্যা
QuickSort মেথড: এটি মূল Quick Sort লজিক প্রয়োগ করে। এটি low এবং high ইনডেক্স গ্রহণ করে, এবং যদি low < high হয়, তখন পার্টিশনিং করা হয় এবং বাম ও ডান অংশে Quick Sort পুনরায় প্রয়োগ করা হয়।

Partition মেথড: এটি পিভট নির্বাচন করে এবং সারণীটি পুনর্বিন্যাস করে যাতে পিভটের ছোট এলিমেন্টগুলি পিভটের বামে এবং বড় এলিমেন্টগুলি পিভটের ডানে চলে যায়।

Swap মেথড: এটি দুটি এলিমেন্টের মান পরিবর্তন করে।

PrintArray মেথড: এটি সারণীর সকল এলিমেন্ট কনসোলে প্রদর্শন করে।

এই কোডটি একটি অ্যারে সোর্ট করতে Quick Sort অ্যালগ

    using System;

    class Program
    {
        static void Main()
        {
            int[] array = { 34, 7, 23, 32, 5, 62, 32, 32 };
            
            Console.WriteLine("Before Sorting:");
            PrintArray(array);
            
            QuickSort(array, 0, array.Length - 1);
            
            Console.WriteLine("After Sorting:");
            PrintArray(array);
        }
    
        // Quick Sort মেথড
        static void QuickSort(int[] arr, int low, int high)
        {
            if (low < high)
            {
                int pi = Partition(arr, low, high);
                
                QuickSort(arr, low, pi - 1);  // বাম অংশে সোর্টিং
                QuickSort(arr, pi + 1, high); // ডান অংশে সোর্টিং
            }
        }
    
        // পার্টিশন মেথড
        static int Partition(int[] arr, int low, int high)
        {
            int pivot = arr[high];  // পিভট হিসেবে শেষ এলিমেন্ট নির্বাচন
            int i = (low - 1);  // ছোট এলিমেন্টের ইনডেক্স
    
            for (int j = low; j <= high - 1; j++)
            {
                // যদি বর্তমান এলিমেন্ট পিভট থেকে ছোট হয়
                if (arr[j] < pivot)
                {
                    i++;
                    Swap(ref arr[i], ref arr[j]);
                }
            }
            Swap(ref arr[i + 1], ref arr[high]);
            return (i + 1);
        }
    
        // দুটি এলিমেন্টের মান বদলানোর মেথড
        static void Swap(ref int a, ref int b)
        {
            int temp = a;
            a = b;
            b = temp;
        }
    
        // অ্যারে প্রিন্ট করার মেথড
        static void PrintArray(int[] arr)
        {
            foreach (var item in arr)
            {
                Console.Write(item + " ");
            }
            Console.WriteLine();
        }
    }

Problem 5: <-------STACK DATA STRUCTURE EXAMPLE IN C# (BANGLA EXPLAIN) ----> 


Stack ডেটা স্ট্রাকচার হলো একটি লাস্ট-ইন-ফার্স্ট-আউট (LIFO) ডেটা স্ট্রাকচার, যেখানে শেষ যে আইটেমটি যুক্ত করা হয় সেটি প্রথমে অপসারণ করা হয়। স্ট্যাকের প্রধান দুটি অপারেশন হলো:

Push: নতুন একটি আইটেম স্ট্যাকে যুক্ত করা।
Pop: সর্বশেষ যুক্তকৃত আইটেমটি স্ট্যাক থেকে সরিয়ে ফেলা।

    using System;

    class Stack
    {
        private int[] elements;   // স্ট্যাকের জন্য একটি অ্যারে ব্যবহার করা হচ্ছে
        private int top;          // স্ট্যাকের শীর্ষ পয়েন্টার
        private int max;          // স্ট্যাকের সাইজ
    
        // Constructor to initialize the stack
        public Stack(int size)
        {
            elements = new int[size];  // স্ট্যাকের জন্য নির্ধারিত সাইজের অ্যারে তৈরি
            top = -1;                  // স্ট্যাকের শীর্ষ পয়েন্টার শুরুতে খালি থাকে
            max = size;                // সর্বোচ্চ সাইজ
        }
    
        // Push operation: নতুন আইটেম স্ট্যাকে যুক্ত করা
        public void Push(int item)
        {
            if (top == max - 1)
            {
                Console.WriteLine("Stack Overflow");  // যদি স্ট্যাক পূর্ণ থাকে
            }
            else
            {
                elements[++top] = item;               // শীর্ষ পয়েন্টার বাড়িয়ে নতুন আইটেম যোগ করা
                Console.WriteLine($"Pushed {item} into the stack.");
            }
        }
    
        // Pop operation: সর্বশেষ আইটেম স্ট্যাক থেকে সরিয়ে ফেলা
        public int Pop()
        {
            if (top == -1)
            {
                Console.WriteLine("Stack Underflow"); // যদি স্ট্যাক খালি থাকে
                return -1;
            }
            else
            {
                int poppedElement = elements[top--];  // শীর্ষ থেকে আইটেম সরানো এবং পয়েন্টার কমানো
                Console.WriteLine($"Popped {poppedElement} from the stack.");
                return poppedElement;
            }
        }
    
        // Display the current stack elements
        public void Display()
        {
            if (top == -1)
            {
                Console.WriteLine("Stack is empty");
            }
            else
            {
                Console.WriteLine("Stack elements are:");
                for (int i = top; i >= 0; i--)
                {
                    Console.WriteLine(elements[i]);
                }
            }
        }
    }
    
    class Program
    {
        static void Main(string[] args)
        {
            Stack stack = new Stack(5);  // সাইজ ৫ এর একটি স্ট্যাক তৈরি
    
            stack.Push(10);  // স্ট্যাকে ১০ পুশ
            stack.Push(20);  // স্ট্যাকে ২০ পুশ
            stack.Push(30);  // স্ট্যাকে ৩০ পুশ
            
            stack.Display(); // স্ট্যাকের বর্তমান এলিমেন্টগুলো দেখানো
            
            stack.Pop();     // স্ট্যাক থেকে পপ করা
            stack.Pop();     // স্ট্যাক থেকে পপ করা
            
            stack.Display(); // আবার স্ট্যাকের এলিমেন্টগুলো দেখানো
        }
    }

Problem 6: <-------QUEUE DATA STRUCTURE EXAMPLE IN C# (BANGLA EXPLAIN) ----> 

কিউ (Queue) ডেটা স্ট্রাকচার একটি লিনিয়ার ডেটা স্ট্রাকচার যা FIFO (First In, First Out) প্রিন্সিপাল অনুসরণ করে। অর্থাৎ, যেটি প্রথমে প্রবেশ করে সেটি প্রথমে বের হয়। কিউ ব্যবহার করে বিভিন্ন ধরনের ডেটা প্রক্রিয়া করা যায়, যেমন প্রিন্টিং জব পরিচালনা, প্রক্রিয়া সময়সূচী, ইত্যাদি।

নিচে C# এ কিউ ডেটা স্ট্রাকচার ইমপ্লিমেন্ট করার বিস্তারিত ব্যাখ্যা দেওয়া হলো।

১. মৌলিক কিউ ডেটা স্ট্রাকচার
এখানে আমরা একটি মৌলিক কিউ ক্লাস তৈরি করব যা এনক্যাপসুলেটেড পদ্ধতিতে মৌলিক কিউ কার্যকারিতা প্রদান করবে।
ব্যাখ্যা
একটি নতুন কিউ তৈরি করা হয়েছে এবং তিনটি সংখ্যা কিউতে যুক্ত করা হয়েছে।
Peek মেথড ব্যবহার করে প্রথম আইটেম দেখা হয়েছে।
একটি লুপ ব্যবহার করে সব আইটেম কিউ থেকে সরানো হয়েছে।
এই মৌলিক কিউ ক্লাসটি C# এ কিউ ডেটা স্ট্রাকচার ইমপ্লিমেন্ট করার জন্য একটি সাধারণ উদাহরণ। আপনি এই ক্লাসটিকে আরো উন্নত করতে পারেন যেমন কিউ সাইজ সীমিত করা বা মাল্টিথ্রেডিং সমর্থন যুক্ত করার মাধ্যমে।

    using System;
    using System.Collections.Generic;
    
    public class Queue
    {
        private LinkedList _items = new LinkedList();
    
        // এনক্যাপসুলেটেড পদ্ধতি - আইটেম কিউতে যুক্ত করার জন্য
        public void Enqueue(T item)
        {
            _items.AddLast(item);
        }
    
        // এনক্যাপসুলেটেড পদ্ধতি - কিউ থেকে আইটেম সরানোর জন্য
        public T Dequeue()
        {
            if (IsEmpty())
            {
                throw new InvalidOperationException("Queue is empty");
            }
            T value = _items.First.Value;
            _items.RemoveFirst();
            return value;
        }
    
        // এনক্যাপসুলেটেড পদ্ধতি - কিউয়ের প্রথম আইটেম দেখার জন্য
        public T Peek()
        {
            if (IsEmpty())
            {
                throw new InvalidOperationException("Queue is empty");
            }
            return _items.First.Value;
        }
    
        // এনক্যাপসুলেটেড পদ্ধতি - কিউটি খালি কিনা পরীক্ষা করার জন্য
        public bool IsEmpty()
        {
            return _items.Count == 0;
        }
    
        // কিউয়ের আইটেম সংখ্যা দেখানোর জন্য
        public int Count
        {
            get { return _items.Count; }
        }
    }




    public class Program
{
    public static void Main()
    {
        Queue queue = new Queue();

        queue.Enqueue(1);
        queue.Enqueue(2);
        queue.Enqueue(3);

        Console.WriteLine("Front item: " + queue.Peek()); // Output: Front item: 1

        while (!queue.IsEmpty())
        {
            Console.WriteLine(queue.Dequeue()); // Output: 1, 2, 3
        }
    }
}

Problem 7: <-------MAXIMUM CHAR OCCURING EXAMPLE IN C# (BANGLA EXPLAIN) ----> 


            


    using System;
    using System.Collections.Generic;
    
    class Program
    {
        static void Main()
        {
            string input = "example stringaaaaaaaaaaaaaa";
            var result = GetMaxOccurringChar(input);
            Console.WriteLine($"The character that appears the most is: '{result.Item1}' with a count of {result.Item2}");
        }
    
        static Tuple GetMaxOccurringChar(string input)
        {
            // Create a dictionary to store the count of each character
            Dictionary charCount = new Dictionary();
    
            foreach (char c in input)
            {
                if (charCount.ContainsKey(c))
                {
                    charCount[c]++;
                }
                else
                {
                    charCount[c] = 1;
                }
            }
    
            // Variables to keep track of the character with the maximum count
            char maxChar = '\0';  // Initialize with a null character
            int maxCount = 0;     // To store the maximum occurrence count
    
            // Iterate over the dictionary to find the character with the maximum count
            foreach (var pair in charCount)
            {
                if (pair.Value > maxCount)
                {
                    maxCount = pair.Value;
                    maxChar = pair.Key;
                }
            }
    
            // Return both the character and the maximum count
            return new Tuple(maxChar, maxCount);
        }
    }
